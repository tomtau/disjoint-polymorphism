type Comment = { content : T -> String };

trait comment(content : String) { self : Comment =>
  def content() = content;
};

type Up = { upvotes : T -> Int };

trait up(upvotes : Int) { self : Up =>
  def upvotes() = upvotes;
};

def test = new[Comment & Up] comment("Have fun!") & up(4);

test.content()
