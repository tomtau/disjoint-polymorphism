type EvenOdd = {
  even : Double -> Bool,
  odd : Double -> Bool
}

trait even() { self : EvenOdd =>
  def even(n : Double)  = if n == 0 then true else self.odd(n - 1)
}

trait odd() { self : EvenOdd =>
  def odd(n : Double)  = if n == 0 then false else self.even(n - 1)
}

main = (new[EvenOdd] even() & odd()).even(42)
